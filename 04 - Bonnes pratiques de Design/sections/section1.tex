\begin{frame}{Transistor MOSFET P-Channel (PMOS)}
    \begin{columns}
        \begin{column}{0.75\textwidth}
            \begin{center}
                $V_{gs}$ négatif!\\
                \vspace{6pt}
                $V_{gs} < -V_t$\\
                \vspace{8pt}
                Faire attention au $V_{gs_{max}}$
            \end{center}
            \vspace{24pt}

            \begin{columns}
                \begin{column}{0.5\textwidth}
                    %\textbf{Branchement normal}
                    \begin{itemize}
                        \item $V_G = \SI{0}{\volt}$
                        \item $V_{gs} = -VDD$
                        \item $-VDD < -V_t$
                        \item Conduit!
                    \end{itemize}
                \end{column}
                \begin{column}{0.5\textwidth}
                    %\textbf{Branchement inverse}
                    \begin{itemize}
                        \item $V_G = VDD$
                        \item $V_{gs} = \SI{0}{\volt}$
                        \item $\SI{0}{\volt} > -V_t$
                        \item Ne conduit pas
                    \end{itemize}
                \end{column}
            \end{columns}
        \end{column}
